            \item $\db{\sum_{n\ge2}x^n\log(n)=}\sum_{n\ge2}a_n\cdot(x-x_0)^n$

                  a_n=\log(n)\\
                  x_0=0
            \end{array}\qquad\varphi=\lim_{n\to\infty}\left|\dfrac{a_n}{a_{n+1}}\right|=\lim_{n\to\infty}\left|
